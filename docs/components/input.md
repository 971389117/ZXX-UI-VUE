# Input

<ZXInput></ZXInput>

```html
<!--最普通的输入框-->
<z-input placeholder="请输入内容"></z-input>

<!--带清空按钮的输入框-->
<z-input type="clear" placeholder="请输入内容"></z-input>

<!--密码输入框-->
<z-input type='password' placeholder="请输入内容"></z-input>

<!--图标 - 搜索框-->
<z-input icon="search" placeholder="请输入内容"></z-input>

<!--图标 - 右侧-->
<z-input icon="search" icon-position="right" placeholder="图标在右边"></z-input>

<!--只读-->
<z-input icon="search" readonly icon-position="right" placeholder="只读框"></z-input>

<!--禁用-->
<z-input icon="search" disabled icon-position="right" placeholder="禁用框"></z-input>
```
